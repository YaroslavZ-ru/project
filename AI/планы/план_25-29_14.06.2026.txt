ДЕТАЛИ РЕАЛИЗАЦИИ src/metrics.py
----------------------------------
На уровне модуля (опциональный prometheus_client):
  _PROMETHEUS_AVAILABLE = True
  try:
      import prometheus_client
      from prometheus_client import Counter, Histogram, Gauge, generate_latest
  except ImportError:
      _PROMETHEUS_AVAILABLE = False

Класс MetricsCollector:
  НЕ является синглтоном. Создаётся один раз в _init_components.

  Конструктор __init__(self, use_metrics: bool = False):
    self._use_metrics   = use_metrics
    self._prometheus_on = use_metrics and _PROMETHEUS_AVAILABLE
    Если use_metrics and not _PROMETHEUS_AVAILABLE:
      logger.warning("prometheus_client не установлен. Метрики — только внутренние.")

    Внутренние счётчики (всегда активны, независимо от Prometheus):
      self._counts: dict[str, int] = {
          "requests_total":    0,
          "requests_ok":       0,
          "requests_fallback": 0,
          "requests_error":    0,
          "cache_hits":        0,
          "cache_misses":      0,
      }
      self._total_duration: float = 0.0   (сумма времён в секундах)
      self._lock = threading.Lock()       (для потокобезопасности)

    Если self._prometheus_on:
      Зарегистрировать Prometheus-метрики. Использовать registry=None
      (дефолтный global registry) только ЕСЛИ не уже зарегистрированы.
      ИЛЛЮСТРАЦИЯ (безопасная регистрация):
        try:
            self._prom_requests_total = Counter(
                "ait_requests_total", "Всего запросов AI-Terminator",
                ["status"]
            )
            self._prom_duration = Histogram(
                "ait_request_duration_seconds", "Время обработки запроса"
            )
            self._prom_cache_hits = Counter(
                "ait_cache_hits_total", "Попаданий в кэш векторов"
            )
        except ValueError:
            # Метрики уже зарегистрированы (повторное создание в тестах)
            self._prometheus_on = False

  Метод record_request(duration_s: float, status: str) -> None:
    status — одно из: "ok", "fallback", "error".
    С self._lock:
      self._counts["requests_total"]    += 1
      self._counts["requests_" + status] = self._counts.get(
          "requests_" + status, 0) + 1
      self._total_duration += max(0.0, duration_s)
    Если self._prometheus_on:
      self._prom_requests_total.labels(status=status).inc()
      self._prom_duration.observe(duration_s)
    logger.debug("metrics.record_request: status=%s, duration=%.3fs", status, duration_s)

  Метод record_cache_hit() -> None:
    С self._lock: self._counts["cache_hits"] += 1
    Если self._prometheus_on: self._prom_cache_hits.inc()

  Метод record_cache_miss() -> None:
    С self._lock: self._counts["cache_misses"] += 1

  Метод get_summary() -> dict:
    С self._lock:
      total = self._counts["requests_total"]
      avg_dur = (self._total_duration / total) if total > 0 else 0.0
      return {
          "requests_total":    total,
          "requests_ok":       self._counts["requests_ok"],
          "requests_fallback": self._counts["requests_fallback"],
          "requests_error":    self._counts["requests_error"],
          "cache_hits":        self._counts["cache_hits"],
          "cache_misses":      self._counts["cache_misses"],
          "avg_duration_s":    round(avg_dur, 4),
          "prometheus_active": self._prometheus_on,
      }

  Метод get_prometheus_text() -> str | None:
    Если not self._prometheus_on: return None.
    try:
        return generate_latest().decode("utf-8")
    except Exception as exc:
        logger.error("Ошибка generate_latest: %s", exc)
        return None

Lifespan context manager:
  @asynccontextmanager
  async def lifespan(app: FastAPI):
    global _cfg, _lemmatizer, _synonym_dict, _embedding_model
    global _vector_cache, _kb, _generative_expander, _session_manager, _metrics

    # Startup
    PROJECT_ROOT = Path(__file__).parent.parent
    _cfg = Config.from_json("configs/config.json", project_root=PROJECT_ROOT)
    logger.info("API: конфигурация загружена")

    (_synonym_dict, _lemmatizer, _embedding_model,
     _vector_cache, _kb, _generative_expander,
     _session_manager) = _api_init_components(_cfg)

    _metrics = MetricsCollector(use_metrics=_cfg.use_metrics)
    logger.info("API: компоненты инициализированы")

    # Прогрев модели
    try:
        _ = _embedding_model.get_word_vector("а")
        logger.info("API: прогрев FastText завершён")
    except Exception as exc:
        logger.warning("API: прогрев завершился с ошибкой: %s", exc)

    yield  # приложение работает

    # Shutdown
    if _kb:
        _kb.close()
        logger.info("API: KnowledgeBase закрыт")

  app = FastAPI(
      title="AI-Terminator",
      description="Интеллектуальный помощник по терминам",
      version="1.0.0",
      lifespan=lifespan,
  )

ОБРАБОТКА ОШИБОК:
  - Если компоненты не инициализированы (lifespan ещё не завершился):
    raise HTTPException(503, detail="Сервис запускается. Попробуйте позже.")
  - Если term пустой/невалидный: Pydantic вернёт 422 автоматически.
  - Если run_pipeline вернул {"status": "error"}: вернуть HTTP 200 с этим JSON.
    (Это не HTTP-ошибка, это бизнес-ошибка — пустой term и т.д.)
  - Исключения внутри pipeline: HTTP 500.

ВАЖНО ПРО ПОТОКИ:
  FastTextWrapper и SQLite соединение не потокобезопасны (или условно безопасны).
  Для продакшн-нагрузки: запускать uvicorn с --workers 1 (один процесс).
  При необходимости масштабирования — использовать asyncio.to_thread() для
  вызова blocking ML-кода. В текущем плане достаточно одного воркера.

  Returns:

       conn.row_factory = sqlite3.Row
    2. Получить все концепты:

    3. Для каждого концепта получить параметры:
         params = conn.execute(
             "SELECT name, label_ru, type, description, unit, enum_values, confidence"
             " FROM parameters WHERE concept_id=? ORDER BY id",
             (concept_id,)
         ).fetchall()
    4. Собрать список концептов в формате, совместимом с update_kb.py:

                "description": "...", "unit": "мм", "enum_values": null,
                "confidence": 1.0}
             ]
           },
           ...
         ]
    5. enum_values: если строка — попытаться json.loads(); если не JSON — оставить строку.
    6. Создать output_path.parent.mkdir(parents=True, exist_ok=True)
    7. output_path.write_text(json.dumps(result, ensure_ascii=False, indent=2),
                              encoding="utf-8")
    8. conn.close()
    9. logger.info("Экспорт: %d концептов, %d параметров -> %s",
                   concepts_count, params_count, output_path)
